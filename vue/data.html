# 数据响应式

Vue2 的核心就是数据响应式，当我们的数据发生变化时，dom 会自动更新，这也就是 mvvm 框架中的 vm 层所做的功能，原理主要是通过拦截 Object.defineProperty 中的 getter 和 setter 函数来处理的。 我们在上一章初始化的流程中，主要是执行了_init 方法，其中 initState 主要就是做整个的数据响应式的功能。

# 流程

我们有一个 demo，2s 过后,新的 message 值就会更新为hello vue2展示在界面

new Vue({
	el: "#app",
	template: "<div>{{message}}</div>",
	mounted() {
		setTimeout(() => {
			this.message = "hello vue2";
		}, 2000);
	},
	data: {
		message: "hello vue"
	}
});

initState 函数定义在 core/instance/state.js

这个函数初始化了包括 props，methods,data,computed,watch 等方法和属性,我们主要关心 initData

export function initState(vm: Component) {
	vm._watchers = [];
	const opts = vm.$options;
	if (opts.props) initProps(vm, opts.props);
	if (opts.methods) initMethods(vm, opts.methods);
	if (opts.data) {
		initData(vm);
	} else {
		observe((vm._data = {}), true /* asRootData */);
	}
	if (opts.computed) initComputed(vm, opts.computed);
	if (opts.watch && opts.watch !== nativeWatch) {
		initWatch(vm, opts.watch);
	}
}

initData 的定义也在当前这个文件,除了一些重复属性名称的警告和属性值的访问代理外，主要的就是 observe 函数

function initData(vm: Component) {
	if (props && hasOwn(props, key)) {
		process.env.NODE_ENV !== "production" &&
			warn(
				`The data property "${key}" is already declared as a prop. ` +
					`Use prop default value instead.`,
				vm
			);
	} else if (!isReserved(key)) {
		// 代理，当访问vm.message时候，就会访问vm._data.message
		proxy(vm, `_data`, key);
	}
	// 响应式
	observe(data, true /* asRootData */);
}

observe 函数定义在 core/observer/index.js

一开始会判断传入的 value 是否为对象，如果不是则直接返回 undefined ， 由于 data 的数据是对象，默认会注入__ob__,表明为响应式的对象，该值通过 new Observer 来获取,作为一个观察者实例

export function observe(value: any, asRootData: ?boolean) {
	// 观察者
	let ob: Observer | void;
	if (hasOwn(value, "__ob__") && value.__ob__ instanceof Observer) {
		ob = value.__ob__;
	} else if (
		shouldObserve &&
		!isServerRendering() &&
		(Array.isArray(value) || isPlainObject(value)) &&
		Object.isExtensible(value) &&
		!value._isVue
	) {
		// 返回一个Observer实例
		ob = new Observer(value);
	}
	if (asRootData && ob) {
		ob.vmCount++;
	}
	return ob;
}

Observer 构造函数定义在当前文件

export class Observer {
	value: any;
	dep: Dep;
	vmCount: number; // number of vms that have this object as root $data

	constructor(value: any) {
		this.value = value;
		// 如果object有新增或者删除，或者array中有变更方法，可以通知
		this.dep = new Dep();
		this.vmCount = 0;
		// 设置一个__ob__属性引用当前的Observer实例
		// 不能直接value.__ob__,因为下面会进行for循环进行defineProperty,用了def可以禁止枚举
		def(value, "__ob__", this);
		if (Array.isArray(value)) {
			// 替换数组对象原型
			if (hasProto) {
				protoAugment(value, arrayMethods);
			} else {
				copyAugment(value, arrayMethods, arrayKeys);
			}
			// 如果数组还是对象，也需要响应化处理
			this.observeArray(value);
		} else {
			this.walk(value);
		}
	}

	walk(obj: Object) {
		const keys = Object.keys(obj);
		for (let i = 0; i < keys.length; i++) {
			defineReactive(obj, keys[i]);
		}
	}
}

这里需要考虑传入的 value 是对象还是数组两种情况，如果是对象，则遍历整个对象调用 defineReactive 方法，也就是 Object.defineProperty,这里就是最终需要处理的 getter 和 setter

这个函数也定义在当前文件

export function defineReactive(
	obj: Object,
	key: string,
	val: any,
	customSetter?: ?Function,
	shallow?: boolean
) {
	// 和key一一对应
	const dep = new Dep();

	// 判断值是否依旧是对象
	let childOb = !shallow && observe(val);
	Object.defineProperty(obj, key, {
		enumerable: true,
		configurable: true,
		get: function reactiveGetter() {
			const value = getter ? getter.call(obj) : val;
			// 收集依赖
			if (Dep.target) {
				dep.depend();
				// 如果存子ob if (childOb) {
					childOb.dep.depend();
					if (Array.isArray(value)) {
						dependArray(value);
					}
				}
			}
			return value;
		},
		set: function reactiveSetter(newVal) {
			const value = getter ? getter.call(obj) : val;
			/* eslint-disable no-self-compare */
			if (newVal === value || (newVal !== newVal && value !== value)) {
				return;
			}
			/* eslint-enable no-self-compare */
			if (process.env.NODE_ENV !== "production" && customSetter) {
				customSetter();
			}
			// #7981: for accessor properties without setter
			if (getter && !setter) return;
			if (setter) {
				setter.call(obj, newVal);
			} else {
				val = newVal;
			}
			// 新值也得响应式
			childOb = !shallow && observe(newVal);
			// 通知更新
			dep.notify();
		}
	});
}

# 依赖收集

我们在 data 中定义的每一个 key，都会形成一个 dep 来管理其依赖，如果有子元素也是 object，同时也会将子元素的 dep 加入管理依赖

data: {
	obj: {
		fn: "";
	}
}

// 收集了同一个Watcher,也就是渲染Watcher
// 渲染Watcher的回调函数就是更新dom vm._update(vm._render(), hydrating)
objDep: [Watcher];
fnDep: [Watcher];

// 这个的Dep.target就是渲染Watcher，Vue2中每一个组件就有一个渲染Watcher
if (Dep.target) {
	// 收集依赖,将Watcher收集，作为数据变化的订阅者
	dep.depend();
	// 如果存子ob
	if (childOb) {
		childOb.dep.depend();
		if (Array.isArray(value)) {
			dependArray(value);
		}
	}
}

在 watcher.js 中，主要执行的是 addDep 方法中的 addSub 来添加 Watcher

addDep (dep: Dep) {
  const id = dep.id
  if (!this.newDepIds.has(id)) {
    this.newDepIds.add(id)
    this.newDeps.push(dep)
    if (!this.depIds.has(id)) {
			// 关键的步骤：添加watcher到subs数组中
			dep.addSub(this)
    }
  }
 }

每次收集完依赖之后，还需要做一次清除依赖的工作,有个比较关键的判断

  1. newDeps: 每次 getter 都存入新的 dep，处理完成后将其赋给 deps 作为保留，自身清空
  2. deps: 每次都存之前的 dep，如果说 newDeps 新增了之前没有这个 dep，则删除这个 dep 的当前 Watcher

这么做的一个好处是，如果说之前的某个 dep 比如被 v-if 给隐藏了，就不需要在执行 Watcher 的回调函数，优化性能。

cleanupDeps(){
	// 第一次空，不会执行循环
  let i = this.deps.length
  while (i--) {
    const dep = this.deps[i]
    if (!this.newDepIds.has(dep.id)) {
      dep.removeSub(this)
    }
  }
}

# 数据更新

当数据放生变化后，通知所有订阅这个数据变化的 watcher 执行 update 操作

  1. 当数据发生变化，会调用 setter 函数，Vue 会开启一个队列，缓存同一事件循环的数据变化
  2. 将 watcher 推入到队列。
  3. Vue 在内部通过异步的 Promise.then,MutationObserver,setImmediate 来执行对应的 wacher 函数

# 总结

  1. initData 中的 observe(data)数据响应化处理
  2. observe 方法返回一个 Observer 实例
  3. Observer 根据数组或者对象进行响应化处理，getter 作为依赖收集，setter 作为派发更新
  4. Dep: 管理一组 Watcher
  5. Watcher: 解析一个表达式并且收集依赖，当数据变化时候触发回调函数
  6. 每个组件有个渲染 Watcher，作为数据变化的订阅者，当数据变化时候会触发 update 方法重新渲染
