# Vue2.x 源码学习

# 1.当我们 new Vue 的时候，发生了什么？

一般我们用 vue 进行开发的时候，入口处基本都是类似的语法:

new Vue({
	el: "#app",
	template: "<div>{{message}}</div>",
	data: {
		message: "hello vue"
	}
});

用过 vue 开发的都知道，这个时候在页面中应该会有一段 div，里面的内容就是 hello vue。那为什么数据会通过变量的形式展示在页面中呢？我们来分析下这个流程。

我们去 github 下载 vue 的源码，主要看 vue 的核心目录结构:

src
├── compiler  # 编译
├── core      # 核心
├── platforms # 各个平台
├── server    # ssr
├── sfc       # vue文件处理
└── shared    # 共享的工具等

我们从整体看，整个项目主要采用了 rollup 来进行构建，将这些逻辑打包成不同规范的文件。 vue 的构建一般分为带 compiler 版本和不带 compiler 的 runtime 版本，不带 compiler 的一般是由 webpack 等工具来处理编译的过程，整体的大小会小很多,由于我们在初始化的时候带有 template 标签，所以自然使用带编译的版本来看代码。

我们直接看scripts目录下的config.js中的某个配置:

 web-full-cjs-dev': {
    entry: resolve('web/entry-runtime-with-compiler.js'),
    dest: resolve('dist/vue.common.dev.js'),
    format: 'cjs',
    env: 'development',
    alias: { he: './entity-decoder' },
    banner
  }

这个 entry 就是打包的文件入口，包含了 comipler 的内容,也就是我们要看的第一个文件。

终于找到了最后的定位内容，一个构造函数 Vue 的定义，也就是我们在入口处实例化的内容,找到这个函数之后，我们就可以慢慢的去看，框架到底是如何去渲染内容到页面的。

首先我们看 new Vue 的时候，会初始化执行 this._init(options),同时将我们的参数传入,我们这边传入了 3 个参数 el,template,data到原型的 init 方法,我们看下 init 方法。

init 的注册是在src/core/instance/init,通过一个函数initMixin扩展构造函数的原型方法_init。 我们省略了不需要这次看的代码，只看这次执行的过程,

先看关键的函数: resolveConstructorOptions:

// `resolveConstructorOptions(vm.constructor)`
// 这个函数会传入实例的构造函数，也就是源码定义的 Vue:
export function resolveConstructorOptions(Ctor: Class<Component>) {
	// 首先获取Vue函数上的options
	// 在src/core/index中， 通过initGlobal扩展Vue,主要添加了components,filters,directive.同时将自身存在_base
	let options = Ctor.options;
	// 这个例子不存在基类,直接跳过返回
	if (Ctor.super) {
		//
	}
	// options = {
	//   components: {KeepAlive: {…}, Transition: {…}, TransitionGroup: {…}}
	//   directives: {model: {…}, show: {…}}
	//   filters: {}
	//   _base: ƒ Vue(options)
	// }
	return options;
}

接着看函数: mergeOptions

// mergeOptions(parent,child,vm)
// 这个函数主要是合并所有的配置
export function mergeOptions(
	parent: Object,
	child: Object,
	vm?: Component
): Object {
	// 判断是否有components，目前没有
	// 对props，inject，directives进行序列化
	// 处理extends和mixins

	// 下面相当于不同属性的合并策略，这边针对parent，也就是components，filters,directives的策略定义在当前文件的mergeAssets中,_base会走默认的strats。
	// strat(parent[key], child[key], vm, key)
	// 这里会有个判断，如果child内有对应设置的key，就会进行一个覆盖，否则直接返回
	/*
  function mergeAssets(
    parentVal: ?Object,
    childVal: ?Object,
    vm?: Component,
    key: string
  ): Object {
    const res = Object.create(parentVal || null);
    if (childVal) {
      process.env.NODE_ENV !== "production" &&
        assertObjectType(key, childVal, vm);
      return extend(res, childVal);
    } else {
      return res;
    }
  }
  */
	const options = {};
	let key;
	for (key in parent) {
		mergeField(key);
	}
	// 这里会进行三个不同属性的合并data,template,el
	// el,template的策略为默认策略，返回child为主，没有则返回parent
	// data的策略为mergeDataOrFn,对于函数会执行并返回,同时会覆盖
	for (key in child) {
		if (!hasOwn(parent, key)) {
			mergeField(key);
		}
	}
	// 不同的属性不同的合并策略
	function mergeField(key) {
		const strat = strats[key] || defaultStrat;
		options[key] = strat(parent[key], child[key], vm, key);
	}
	return options;
}

接下来就是定义的 init 函数，忽略了不需要执行的过程:

export function initMixin(Vue: Class<Component>) {
	Vue.prototype._init = function(options?: Object) {
		// 实例存储
		// 每个实例都有递增的_uid
		// 合并配置
		if (options && options._isComponent) {
			// 每个子组件初始化配置合并，做了性能优化
		} else {
			// mergeOptions的功能主要是为了合并Vue的一些全局配置到我们传入的配置
			vm.$options = mergeOptions(
				resolveConstructorOptions(vm.constructor),
				options || {},
				vm
			);
		}
		// 开发环境做一层拦截，遇到没有定义过的数据会报错
		/* istanbul ignore else */
		if (process.env.NODE_ENV !== "production") {
			initProxy(vm);
		} else {
			vm._renderProxy = vm;
		}

		// 一些初始化和钩子的执行
		// expose real self
		vm._self = vm;
		initLifecycle(vm);
		initEvents(vm);
		initRender(vm);
		callHook(vm, "beforeCreate");
		initInjections(vm); // resolve injections before data/props
		initState(vm);
		initProvide(vm); // resolve provide after data/props
		callHook(vm, "created");

		// 我们的options有传入el，所以会调用$mount方法
		if (vm.$options.el) {
			vm.$mount(vm.$options.el);
		}
	};
}

$mount方法定义在入口src/platforms/web/entry-runtime-with-compiler

Vue.prototype.$mount = function(
  el?: string | Element,
  hydrating?: boolean
): Component {
  // el做一层处理，如果有则获取，如果没有则会报错并且创建dom
  el = el && query(el);

  // 不允许挂在body和html中
  /* istanbul ignore if */
  if (el === document.body || el === document.documentElement) {
    process.env.NODE_ENV !== "production" &&
      warn(
        `Do not mount Vue to <html> or <body> - mount to normal elements instead.`
      );
    return this;
  }

  const options = this.$options;
  // 我们这边采用了tempalte选项， 最终会将template转化成render函数
  if (!options.render) {
    let template = options.template;
    ...

    if (template) {

      // 编译入口,将template转为render函数
      const { render, staticRenderFns } = compileToFunctions(
        template,
        {
          outputSourceRange: process.env.NODE_ENV !== "production",
          shouldDecodeNewlines,
          shouldDecodeNewlinesForHref,
          delimiters: options.delimiters,
          comments: options.comments
        },
        this
      );
      options.render = render;
      options.staticRenderFns = staticRenderFns;

    }
  }
  // 执行之前缓存的mount函数
  return mount.call(this, el, hydrating);
};

之后会执行src/platforms/web/runtime/index内的 mount 函数,这个比较简单，会调用 mountComponent 方法

// public mount method

mountComponent 方法在src/core/instance/lifecycle中

export function mountComponent(
	vm: Component,
	el: ?Element,
	hydrating?: boolean
): Component {
	// 核心为 new一个Watcher
	new Watcher(
		vm,
		updateComponent,
		noop,
		{
			before() {
				if (vm._isMounted && !vm._isDestroyed) {
					callHook(vm, "beforeUpdate");
				}
			}
		},
		true /* isRenderWatcher */
	);
	hydrating = false;

	// manually mounted instance, call mounted on self
	// mounted is called for render-created child components in its inserted hook
	// vm.$vnode为父虚拟node,表示为根实例
	if (vm.$vnode == null) {
		vm._isMounted = true;
		callHook(vm, "mounted");
	}
	return vm;
}

mountComponent 函数主要定义了一个 updateComponent 方法，然后通过 new Watcher 传入，我们看看 Watcher 做了什么。

在src/core/observer/watcher中定义了 Watcher 类

// 我们省略暂时不需要的代码
export default class Watcher {
  vm: Component;
  expression: string;
  cb: Function;
  id: number;
  deep: boolean;
  user: boolean;
  lazy: boolean;
  sync: boolean;
  dirty: boolean;
  active: boolean;
  deps: Array<Dep>;
  newDeps: Array<Dep>;
  depIds: SimpleSet;
  newDepIds: SimpleSet;
  before: ?Function;
  getter: Function;
  value: any;

  constructor (
    vm: Component,
    expOrFn: string | Function,
    cb: Function,
    options?: ?Object,
    isRenderWatcher?: boolean
  ) {
    if (options) {
    // 这里就会把updateComponet函数赋给this.getter
    if (typeof expOrFn === 'function') {
      this.getter = expOrFn
    } else {
      this.getter = parsePath(expOrFn)
      if (!this.getter) {
        this.getter = noop
        process.env.NODE_ENV !== 'production' && warn(
          `Failed watching path: "${expOrFn}" ` +
          'Watcher only accepts simple dot-delimited paths. ' +
          'For full control, use a function instead.',
          vm
        )
      }
    }
    this.value = this.lazy
      ? undefined
      : this.get()
  }
  get () {
    pushTarget(this)
    let value
    const vm = this.vm
    try {
      value = this.getter.call(vm, vm)
    } catch (e) {
      if (this.user) {
        handleError(e, vm, `getter for watcher "${this.expression}"`)
      } else {
        throw e
      }
    } finally {
      // "touch" every property so they are all tracked as
      // dependencies for deep watching
      if (this.deep) {
        traverse(value)
      }
      popTarget()
      this.cleanupDeps()
    }
    return value
  }
}

这里我们可以很直接的看到，new Watcher 之后，我们会执行 getter 函数，也就是外部定义的 updateComponent。 在回到之前定义 updateComponent 的地方,也就是执行了下面这句话:

// updateComponent函数执行
vm._update(vm._render(), hydrating);

这里涉及到两个函数 vm._update 和 vm._render，第二个参数是服务端渲染相关，暂时不用考虑。

我们先看 vm._render,这个函数定义在src/core/instance/render中

// 这个函数最终返回的是一个vnode
Vue.prototype._render = function(): VNode {
	// 获取当前实例
	const vm: Component = this;
	// 获取属性
	const { render, _parentVnode } = vm.$options;
	vm.$vnode = _parentVnode;
	// render self
	let vnode;
	try {
		currentRenderingInstance = vm;
		// 关键点就是执行render函数
		vnode = render.call(vm._renderProxy, vm.$createElement);
	} catch (e) {
		// ...
	} finally {
		currentRenderingInstance = null;
	}
	// if the returned array contains only a single node, allow it
	if (Array.isArray(vnode) && vnode.length === 1) {
		vnode = vnode[0];
	}
	// return empty vnode in case the render function errored out
	if (!(vnode instanceof VNode)) {
		if (process.env.NODE_ENV !== "production" && Array.isArray(vnode)) {
			warn(
				"Multiple root nodes returned from render function. Render function " +
					"should return a single root node.",
				vm
			);
		}
		vnode = createEmptyVNode();
	}
	// set parent
	vnode.parent = _parentVnode;
	return vnode;
};

这里最终会执行实例上的 render 函数，同时传入第一个参数 vm.$createElement
