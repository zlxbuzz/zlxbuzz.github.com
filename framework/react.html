# React 相关笔记

# 生命周期

# 1.组件存在的过程为一个生命周期

mount->update->unmount

# 2. Effect 存在一个生命周期

创建执行->销毁 组件里的 Effect 的挂载和销毁可能会有多次

function App() {
  const [name] = useState("");

  useEffect(() => {
    console.log("create");
    // 当数据name变化时，Effect总会先销毁，后创建
    return () => {
      console.log("unmount");
    };
  }, [name]);
}

# 内置 hook

useCallback 避免每次 render 重新定义函数

function App() {
  // 会保存这个函数的引用
  const handleCb = useCallback(() => {
    console.log("cb");
  }, []);

  return <div @click={handleCb}> click me </div>
}

useDebugValue 使用在自定义 hook 中，进行输出调试

function useMyhook() {
  useDebugValue("hello");
}
